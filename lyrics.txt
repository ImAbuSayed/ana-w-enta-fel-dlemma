Song
أنا وانتوا في الضلمة (Ana w Enta fel Dlemma) – You and Me in the Dark
Lyrics
[الکوبـــل الأول (Verse 1)]
عـُـيــونـــك نـاسـيـاني .. و الـضـلـمـة جــابـتـنـي
واحـــد أفـــراحـــي مــابـقـاش .. إلا ذكــراك تـعـذّبـنـي
الوقــت قـاسـي يـا حـبـيـبـي .. و القلب صـار يـتـنـدّم
أنا لـمـّـا أشــوف شـبـهــك .. الـدمــع يـغـلـب الـكـلـم
[الکورس (Chorus)]
أنا و إنتــوا فـي الضلمة .. و الـحـبّ مـات مـالـقـيـنـاش
نــارك فـي قـلـبـي مــش بـتـطـفـى .. و الـجـرح لـسـه مـا تـنـاسـيـنـاش
يـا لـيـت الـزّمـان يـرُدّ .. لـحـظـة مـن أيـامـنـا
نـعـيـش فـي دنـيـا مـا فـيـهـاش .. غـيـابـك ولا هـمـومـنـا
[الکوبـــل التـاني (Verse 2)]
الـصـبـح بــدونـك ظـلـمـة .. و الـلـيـل طـولـه عـذاب
كـل مــا أقول أروح أهــون .. الـبـعـد يـزيـد الاحـتـراق
أنـا مـا عـرفـتـش إزاي .. أرضـى بــدونك عـمـرٍي
الـقـلـب نـاديـانـي لـيـك .. و الـحـزن مـسـك إدري
[الجـِسـر (Bridge)]
يـا شـمـعـتـيـن انـطـفـوا .. مـالـقـيـتـش فـيـكـم نـور
الـدنـيـا تـعـبـتـنـي .. و الـحـلـم صـار سـطـور
سـألـت الـنـجـوم تـرُد .. لـمـّـا تـرجع تـفـضـل؟
قالـوا الـغـيـاب قـدّامـنـا .. و الـقـدر مـابـيـسـمـحـش
[الختـام (Outro)]
يــا زمــان .. خــلـيـنـي أنــسـى
واحــشـنـي لـو قـدرت .. و الــدنيــا تــرحمـنـي
بـس يـا قـلـب لا تـنـسـى .. إنــك عـشـت وتـعـبـت
و الــحـبّ مــا بـيـمـوتـش .. لــو جـرحـك مــا اتـعـلـجـت
